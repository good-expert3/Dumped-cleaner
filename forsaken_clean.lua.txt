-- Load WindUI framework from GitHub
local function loadWindUI()
    local httpGet = game:HttpGet()
    local windUIUrl = "https://github.com/Footagesus/WindUI/releases/latest/download/main.lua"
    local windUICode = httpGet(game, windUIUrl)
    local windUIFunction = loadstring(windUICode)
    return windUIFunction()
end

-- Initialize custom audio IDs from file if it exists
local function loadCustomAudioIDs()
    local success, content = pcall(function()
        return readfile("Nullsaken_Audio_Block_IDS.txt")
    end)
    
    if success and content and content ~= "" then
        -- Parse stored custom IDs
        local customIds = content:split(",")
        for _, id in ipairs(customIds) do
            local trimmedId = id:match("^%s*(.-)%s*$")  -- Trim whitespace
            if trimmedId and trimmedId ~= "" then
                genv["NullsakenCustomIDs"][trimmedId] = true
            end
        end
    end
end

-- Load saved custom IDs on startup

    task.spawn(function()
        while staminaToggle.Enabled do
            task.wait(0.005)
            
            -- Reset stamina values to configured amounts
            sprintingModule.MaxStamina = staminaConfig.maxStamina
            sprintingModule.SprintSpeed = staminaConfig.sprintSpeed

            sprintingModule.StaminaDrain = staminaConfig.staminaDrain
            sprintingModule.StaminaRegenDelay = staminaConfig.staminaRegenDelay
            
            -- Set current stamina to maximum
            sprintingModule.Stamina = math.max(0, sprintingModule.MaxStamina)
        end
    end)
end

-- Helper function to find nearest enemy
local function findNearestEnemy()
    local nearestPlayer = nil
    local nearestDistance = math.huge
    
    for _, player in ipairs(players:GetPlayers()) do
        if player ~= localPlayer and player.Character then
            local rootPart = player.Character:FindFirstChild("HumanoidRootPart")
            if rootPart then
                local distance = (rootPart.Position - localPlayer.Character.HumanoidRootPart.Position).Magnitude
                if distance < nearestDistance then
                    nearestDistance = distance
                    nearestPlayer = player
                end
            end
        end
    end
    
    return nearestPlayer
end
